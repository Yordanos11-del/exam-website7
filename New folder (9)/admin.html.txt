    <script>
        // Load question stats
        function loadQuestionStats() {
            const statsContainer = document.getElementById('stats-container');
            if (!statsContainer) return;
            
            const subjects = ['mathematics', 'physics', 'chemistry'];
            let totalQuestions = 0;
            let html = '<div style="display: grid; grid-template-columns: repeat(auto-fit, minmax(200px, 1fr)); gap: 20px; margin-top: 20px;">';
            
            subjects.forEach(subject => {
                const questions = questionDatabase[subject] || [];
                const subjectName = subject.charAt(0).toUpperCase() + subject.slice(1);
                totalQuestions += questions.length;
                
                // Count by difficulty
                const easy = questions.filter(q => q.difficulty === 'easy').length;
                const medium = questions.filter(q => q.difficulty === 'medium').length;
                const hard = questions.filter(q => q.difficulty === 'hard').length;
                
                html += `
                    <div class="stat-card" style="background: #f8f9fa; padding: 20px; border-radius: 10px;">
                        <h4 style="margin-bottom: 10px; color: #667eea;">${subjectName}</h4>
                        <p style="font-size: 2rem; font-weight: bold; margin: 10px 0;">${questions.length}</p>
                        <p style="color: #666; font-size: 0.9rem;">
                            Easy: ${easy} | Medium: ${medium} | Hard: ${hard}
                        </p>
                    </div>
                `;
            });
            
            html += `
                <div class="stat-card" style="background: linear-gradient(45deg, #667eea, #764ba2); color: white; padding: 20px; border-radius: 10px;">
                    <h4 style="margin-bottom: 10px;">Total Questions</h4>
                    <p style="font-size: 2.5rem; font-weight: bold; margin: 10px 0;">${totalQuestions}</p>
                    <p style="opacity: 0.9; font-size: 0.9rem;">
                        All subjects combined
                    </p>
                </div>
            </html>`;
            
            statsContainer.innerHTML = html;
        }
        
        // Initialize admin panel when page loads
        document.addEventListener('DOMContentLoaded', function() {
            loadQuestionStats();
            
            // Update stats every 30 seconds
            setInterval(loadQuestionStats, 30000);
        });
    </script>
